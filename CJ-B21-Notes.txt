import java.util.Arrays;

public class Demo
{
	public static void main( String args[] ) throws Exception
	{
		BufferedReader br  = new BufferedReader( new InputStreamReader(System.in));	

		System.out.print("How many Number : ");
		int n = Integer.parseInt(br.readLine()); // n=3

		int a[] = new int[n];

		/* Array Accepting Logic */

		for(int i=0; i < a.length; i++ )
		{
			System.out.print("Enter the Data :");
			a[i] = Integer.parseInt(br.readLine());
		}

		System.out.println("Data : " +  Arrays.toString(a) );

		/* Sum Calculation Logic */

		int ans = 0;

		for(int i=0; i < a.length; i++)
		{
			ans = ans + a[i];
		}

		System.out.println("Addition of Array Elements : " +  ans );
	}
}

{ 
	private int rno;

	private double per;

	public Student()

		rno  = 0;

		per  = 0.0;

	}

Inheritance with Constructor Calling sequence
==============================================
--> Whenever you create object of child class then automatically child class constructor gets called, to call parent class constructor from child class use super().

--> If 0-arg constructor(Default constructor) does have call to parent constructor then still JVM will by default call Parents Default constructor.

--> In case of Parameterized constructor of child class, you have to call parents parameterized constructor, if you dont call parents constructor explicitely then JVM calls by default parent's 0-arg/default constructor. 

--> super() statment should be first statement constructor.


Example:
========

import java.io.*;

class A
{
	private int i;

	public A()
	{
		i=88;
		System.out.println("A's Default/0-Arg Constructor !!");
	}

	public A(int i)
	{
		this.i = i;
		System.out.println("A's Parameterized Constructor!!");
	}

	public void display()
	{
		System.out.println("I am in A's Display !!");
		System.out.println("Value of I :" + i);
	}
}

class B extends A 
{
	private int j;

	public B()
	{
		super(); // It is optional
		j=99;
		System.out.println("B's Default/0-Arg Constructor !!");

	}

	public B(int i,int j)
	{
		super(i);
		this.j = j;
		System.out.println("B's Parameterized Constructor!!");
	}

	public void display()
	{
		super.display(); // to access private members of Parent
		System.out.println("I am in B's Display !!");
		System.out.println("Value of J :" + j);
	}
}

ArrayList
----------
--> ArrayList is a collection class that implements List interface.
--> It uses a dynamic ARRAY for storing the elements i.e. we do not need to declare its size, it grows as we add elements to it and it shrinks as we remove elements from it, during the runtime of the program. 
--> Note: The ArrayList increases by approximately 50% of its current size whenever it runs out of space.
--> ArrayList class can contain duplicate elements/objects. 
--> ArrayList class maintains insertion order. 
--> â€˜nullâ€™ insertion is possible. 
--> ArrayList class is not synchronized i.e that multiple threads cannot access and modify an ArrayList concurrently. 
--> ArrayList allows "fast random access" because array works at the index basis. 

--> In Java ArrayList class, manipulation is slow because a lot of shifting needs to be occurred if any element is removed from the array list. 

-->  ArrayList is best suitable if our frequent operation is retrieval operation. 

-->  ArrayList is the worst choice if you want to perform insertion or deletion in middle. To perform insertion & deletion go for LinkedList.

Q. ArrayList Vs LinkedList
------------------------
1. Array is Faster in case of accessing elements
	LinkedList is slower in case of accessing element.
	bcz to access 'n-th' element we have to traverse 'n-1' elements.

2. Array is slower in case of deletion and Insertion of an element.
   bcz it does 'n' shift operations for single insertion/deletion.
   In other words - Insertion & deletion in arraya takes O(n) time. 

   LinkedList is Faster in case of insertion or deletion of an element. 
   In other words - Insertion & deletion in linked list takes O(1) time.

Q. What is Comparable & Comparator ?

Q. Are they Similar ?
----------------------
	--> Yes, They are used when we want to sort user-defined objects which are stored in collection.

Q. What is Difference between Them ?
-------------------------------------
--> Using Comparable we can sort() objects only based one Field(rno or name or per)
--> Using Comparator we can sort objects based on any Field(rno or name or per) in any order(Asc or Desc) on user choice.



import java.io.*;
import java.util.*;

class Student 
{
    private int rno;
    private String name;
    private double per;
    
    public Student() 
    { 
    }

    public Student(int rno, String name, double per) 
    {
        this.rno = rno;
        this.name = name;
        this.per = per;
    }
    
    public int getRno()
    {
        return rno;
    }

    public String getName()
    {
        return name;
    }
    
    public double getPer()
    {
        return per;
    }

    public void display()
    {
        System.out.println("Roll Number : " +  rno);
        System.out.println("Name        : " +  name);
        System.out.println("Percentage  : " + per );
    }

    @Override
    public String toString() 
    {
        return "[" + rno + "," + name + "," + per + "]";  // [ 101,AAA,60]
    }
}


class RollNumberAscendingComparator implements Comparator<Student>
{
    public int compare(Student ob1, Student ob2)
    {
        Integer i1 = ob1.getRno();
        Integer i2 = ob2.getRno();

        return i1.compareTo(i2);
    }
}

class RollNumberDescendingComparator implements Comparator<Student>
{
    public int compare(Student ob1, Student ob2)
    {
        Integer i1 = ob1.getRno();
        Integer i2 = ob2.getRno();

        return -i1.compareTo(i2);
    }
}

class NameAscendingComparator implements Comparator<Student>
{
    public int compare(Student ob1, Student ob2)
    {
       String name1 = ob1.getName();
       String name2 = ob2.getName();

       return name1.compareTo(name2); 
    }
}

class NameDescendingComparator implements Comparator<Student>
{
    public int compare(Student ob1, Student ob2)
    {
       String name1 = ob1.getName();
       String name2 = ob2.getName();

       return -name1.compareTo(name2); 
    }
}

class PercentageAscendingComparator implements Comparator<Student>
{
    public int compare(Student ob1, Student ob2)
    {
        Double p1 = ob1.getPer();
        Double p2 = ob2.getPer();

        return p1.compareTo(p2);
    }
}

class PercentageDescendingComparator implements Comparator<Student>
{
    public int compare(Student ob1, Student ob2)
    {
        Double p1 = ob1.getPer();
        Double p2 = ob2.getPer();

        return -p1.compareTo(p2);
    }
}

public class Demo
{
    public static void main(String[] args) 
    {
       
        Student s1 = new Student(103,"Sachin",80);
        Student s2 = new Student(101,"Vijay", 90);
        Student s3 = new Student(102,"Ajay", 40);
               
        List<Student> L = new ArrayList<>();
        L.add(s1);
        L.add(s2);
        L.add(s3);
        
        System.out.println("List of Students Before Sort : "+ L); 

        Collections.sort(L, new RollNumberAscendingComparator());

        System.out.println("Descending Roll number wise : "+ L); 


        Collections.sort(L, new NameAscendingComparator());
        System.out.println("Ascending Namewise : "+ L);

        
        Collections.sort(L, new NameDescendingComparator());
        System.out.println("Descending Namewise : "+ L);


        Collections.sort(L, new PercentageAscendingComparator());
        System.out.println("Ascending Perwise : "+ L);

        Collections.sort(L, new PercentageDescendingComparator());
        System.out.println("Descending Perwise : "+ L);

    }
}
